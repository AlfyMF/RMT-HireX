    UpdatedAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

Table Name : Certification
CREATE TABLE Certification (
    Id CHAR(36)  PRIMARY KEY,
    Name VARCHAR(255) NOT NULL,
    IssuingOrganization VARCHAR(255) NOT NULL,
    Description TEXT,
    CreatedAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UpdatedAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    IsActive BOOLEAN DEFAULT TRUE
    CONSTRAINT certifications_name_unique UNIQUE (Name)
);
CREATE INDEX certifications_issuing_organization_index
    ON certification (IssuingOrganization);

CREATE TABLE Country (
    Id VARCHAR(36) PRIMARY KEY,
    Name VARCHAR(255) NOT NULL UNIQUE,
    Code VARCHAR(3) NOT NULL UNIQUE,
    PhoneCode VARCHAR(5),
    CreatedAt TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    UpdatedAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    IsActive BOOLEAN DEFAULT TRUE
);
-- Index for phoneCode
CREATE INDEX countries_phone_code_index ON Country(PhoneCode);

CREATE TABLE Department (
    Id VARCHAR(36) PRIMARY KEY,
    Name VARCHAR(255) NOT NULL,
    Code VARCHAR(255) NOT NULL,
    Description TEXT,
    DUHead		VARCHAR(36),				
    CDO			VARCHAR(36),
    RecruiterLead	VARCHAR(36),
    CreatedAt TIMESTAMP(0) NOT NULL DEFAULT CURRENT_TIMESTAMP,
    UpdatedAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    IsActive BOOLEAN DEFAULT TRUE,
    CONSTRAINT delivery_units_name_unique UNIQUE (Name),
    CONSTRAINT delivery_units_code_unique UNIQUE (Code)
);
-- Create additional indexes
CREATE INDEX delivery_units_code_index ON Department (Code);

CREATE TABLE JobTitle (
    Id VARCHAR(36) PRIMARY KEY,
    Title VARCHAR(255) NOT NULL UNIQUE,
    Description TEXT,
    Level INTEGER,
    CreatedAt TIMESTAMP(0) NOT NULL DEFAULT CURRENT_TIMESTAMP,
    UpdatedAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    IsActive BOOLEAN DEFAULT TRUE,
);
-- Add index separately
CREATE INDEX job_titles_level_index ON JobTitle(Level);

CREATE TABLE Qualification (
    Id VARCHAR(36) PRIMARY KEY,
    Name VARCHAR(255) NOT NULL UNIQUE,
    Level VARCHAR(255),
    Description TEXT,
    CreatedAt TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    UpdatedAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    IsActive BOOLEAN DEFAULT TRUE
);

-- Index on level column
CREATE INDEX qualifications_level_index ON Qualification(Level);

CREATE TABLE Role (
    Id VARCHAR(36) PRIMARY KEY,
    Name VARCHAR(255) NOT NULL UNIQUE,
    Description TEXT,
    Permissions JSON,
    CreatedAt TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    UpdatedAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    IsActive BOOLEAN DEFAULT TRUE
);

CREATE TABLE Skill (
    Id VARCHAR(36) PRIMARY KEY,
    Name VARCHAR(255) NOT NULL UNIQUE,
    Category VARCHAR(255),
    IsTechnical  BOOLEAN DEFAULT TRUE,
    Importance VARCHAR(20) NOT NULL DEFAULT 'primary' CHECK (Importance IN ('primary', 'secondary', 'nice_to_have')),
    CreatedAt TIMESTAMP WITHOUT TIME ZONE NOT NULL DEFAULT CURRENT_TIMESTAMP,
    UpdatedAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    IsActive BOOLEAN DEFAULT TRUE
);
-- Additional indexes
CREATE INDEX skills_category_is_technical_index ON Skill(Category, "isTechnical");
CREATE INDEX skills_importance_index ON Skill(Importance);
CREATE INDEX skills_is_technical_index ON Skill(IsTechnical);

    IsActive BOOLEAN DEFAULT TRUE

CREATE TABLE WorkShift (
    Id VARCHAR(36) PRIMARY KEY,
    Name VARCHAR(255) NOT NULL,
    StartTime TIME NOT NULL,
    EndTime TIME NOT NULL,
    Description TEXT,
    CreatedAt TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    UpdatedAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    IsActive BOOLEAN DEFAULT TRUE
);
-- Unique constraint on name
CREATE UNIQUE INDEX workshift_name_idx ON WorkShift(Name);
-- Index on startTime and endTime
CREATE INDEX workshift_start_end_idx ON WorkShift(StartTime, EndTime);
